function fvPaddedText(pvText  varchar2,  pnPadToSize   number) 
  return varchar2 is

   vRtnStr                        varchar2(2000) := '';

   function fvChars(pvChar  varchar2, pnLength  number)
     return varchar2 is

      vChars                         varchar2(2000) := '';

   begin
      for i in 1 .. pnLength loop
         vChars := vChars || ' ';
      end loop;
      return vChars;
   end;

begin

   if length(pvText) > pnPadToSize then
      vRtnStr := substr(pvText, 1, pnPadToSize);
   else
      vRtnStr := pvText || fvChars(' ', pnPadToSize - length(pvText));
   end if;

   return vRtnStr;

end fvPaddedText;       

function put_chars                     (pvChar                         IN     varchar2,
                                        pnNum                          IN     number)
  return varchar2 is
   vRtnStr                     varchar2(4000) := '';
begin
   for n in 1 .. pnNum loop
      vRtnStr := vRtnStr || pvChar;
   end loop;

   return vRtnStr;
end put_chars;

function put_sized_text                (pvText                         IN     varchar2,
                                        pnNum                          IN     number,
                                        pvPaddingChar                  IN     varchar2 DEFAULT ' ')
  return varchar2 is
   vRtnStr                     varchar2(4000) := '';
begin
   if length(pvText) >= pnNum then
      return substr(pvText, 1, pnNum);
   else
      return put_chars(pvPaddingChar, pnNum - length(pvText)) || pvText;
   end if;
end put_sized_text;

function fvDecodeNullable              (pvNullableYN                   IN     varchar2)
  return varchar2 is
begin
   if pvNullableYN = 'Y' then
      return null;
   else
      return 'NOT NULL';
   end if;
end fvDecodeNullable;

function fvFormatDataTypes             (pvColumnName                   IN     varchar2,
                                        pvDataType                     IN     varchar2,
                                        pnDataLength                   IN     number,
                                        pvNullableYN                   IN     varchar2,
                                        pnDataPrecision                IN     number,
                                        pnDataScale                    IN     number,
					pvDataDefault                  IN     varchar2)
  return varchar2 is

   vRtnStr                        varchar2(4000);
   vBrackets                      varchar2(100) := null;
   nMaxBracketsLength             number := 16;

begin

   if pvDataType in ('VARCHAR2', 'CHAR') then
      vBrackets := pvDataType || '(' || pnDataLength || ')';
      vBrackets := vBrackets || put_chars(' ', nMaxBracketsLength - length(vBrackets));
      vRtnStr := vRtnStr || chr(10) || put_chars(' ', 6) || pvColumnName || 
                            put_chars(' ', 31 - length(pvColumnName)) || 
                            vBrackets ||
                            fvDecodeNullable(pvNullableYN);

   elsif pvDataType in ('DATE', 'BLOB', 'CLOB') then
      vBrackets := pvDataType;
      vBrackets := vBrackets || put_chars(' ', nMaxBracketsLength - length(vBrackets));
      vRtnStr := vRtnStr || chr(10) || put_chars(' ', 6) || pvColumnName || 
                            put_chars(' ', 31 - length(pvColumnName)) || 
                            vBrackets ||
                            fvDecodeNullable(pvNullableYN);

   elsif pvDataType = 'NUMBER' then
      vBrackets := pvDataType || '(' || pnDataPrecision;
      if nvl(pnDataScale, 0) <> 0 then
         vBrackets := vBrackets || ', ' || pnDataScale || ')';
         vBrackets := vBrackets || put_chars(' ', nMaxBracketsLength - length(vBrackets)) ||
                            fvDecodeNullable(pvNullableYN);
      else
         vBrackets := vBrackets || ')';
         vBrackets := vBrackets || put_chars(' ', nMaxBracketsLength - length(vBrackets)) || 
                            fvDecodeNullable(pvNullableYN);
      end if;

      vRtnStr := vRtnStr || chr(10) || put_chars(' ', 6) || pvColumnName || 
                            put_chars(' ', 31 - length(pvColumnName));
      vRtnStr := vRtnStr || vBrackets;

   end if;

   return vRtnStr;

end fvFormatDataTypes;

function fnObjectNameHash              (pvObjectName                   IN     varchar2)
  return number is

   vObjectName                   varchar2(128) := upper(ltrim(rtrim(pvObjectname)));
   nLetterCode                   number;
   nSum                          number := 0;
   nLetterStartCode              number := ascii('A') - 1;

begin

   for n in 1 .. length(vObjectName) loop
      nLetterCode := ascii(substr(vObjectName, n, 1)) - nLetterStartCode;
      nSum := nSum + nLetterCode;
   end loop;
   return nSum;

exception
   when others then
      return 0;

end fnObjectNameHash;

function get_objects_list              (pvDatabaseName                 IN     varchar2,
                                        pvSchemaName                   IN     varchar2 DEFAULT NULL,
                                        pvObjectType                   IN     varchar2 DEFAULT NULL)
  return varchar2 is

   vDatabaseName                  varchar2(128) := upper(ltrim(rtrim(pvDatabaseName)));
   vSchemaName                    varchar2(128);
   vObjectType                    varchar2(128);

   vStatement                     varchar2(4000);
   vRtnStr                        varchar2(8000) := null;
   vLastObjectType                varchar2(128)  := 'zzz';
   vFirstRow                      varchar2(1)    := 'Y';

   excObjNotFound                 exception;

begin

   if pvSchemaName is null then
      vSchemaName := 'ALL';
   else
      vSchemaName := upper(ltrim(rtrim(pvSchemaName)));
   end if;

   if pvObjectType is null then
      vObjectType := 'ALL';
   else
      vObjectType := upper(ltrim(rtrim(pvObjectType)));
   end if;

   vStatement := 'select do.object_type, do.object_name, do.status, do.object_created, do.object_last_ddl_time, null "RETURN_TYPE" ' ||
                 '  from dba_objects@' || vDatabaseName || ' do ' ||
                 ' where (instr(''' || vObjectType || ''', do.object_type) > 0 or ''' || vObjectType || ''' = ''ALL'')' ||
                 '   and (''' || vSchemaName || ''' = ''ALL'' or do.owner = ''' || vSchemaName || ''')' ||
                 '   and do.object_type in (''TABLE'', ''VIEW'', ''PACKAGE'', ''FUNCTION'', ''PROCEDURE'', ''SEQUENCE'')' ||
                 ' order by decode(do.object_type, ''TABLE'', 1, ''VIEW'', 2, ''PACKAGE'', 3, ''PROCEDURE'', 4, ''FUNCTION'', 5, ''SEQUENCE'', 6), do.object_name';

   vRtnStr := put_chars('=', nLineLength) || chr(10) ||
                 '= REPORT: OBJECTS OWNED BY ' || vSchemaName || '@' || vDatabaseName || 
                 put_chars(' ', nLineLength - 29 - length('= REPORT: OBJECTS OWNED BY ' || vSchemaName || '@' || vDatabaseName)) || 
                 'Run Date: ' || to_char(sysdate, vDateTimeFmt) || chr(10) ||
                 '= ' || chr(10);

   if vObjectType <> 'ALL' then
      vRtnStr := vRtnStr || '= Object Type(s): ' || vObjectType || chr(10);
   end if;

   vRtnStr := vRtnStr || put_chars('-', nLineLength) || chr(10);

   execute immediate vStatement
      bulk collect into object_list_tab;

   if nvl(object_list_tab.last, 0) = 0 then
      raise excObjNotFound;
   end if;

   for i in 1 .. nvl(object_list_tab.last, 0) LOOP

      if length(vRtnStr) > nMaxOutputSizeLimit - 300 then
         vRtnStr := vRtnStr || '- ERROR - Max size hit after ' || i || ' iterations at ' || length(vRtnStr) || ' characters.';
         exit;
      end if;

      if vLastObjectType <> object_list_tab(i).object_type then
         vRtnStr := vRtnStr || chr(10) || object_list_tab(i).object_type || 'S' || chr(10);
         vLastObjectType := object_list_tab(i).object_type;
      end if;

      if vLastObjectType in ('TABLE', 'VIEW') then
         vRtnStr := vRtnStr || put_chars(' ', 3) || object_list_tab(i).object_name ||

      elsif vLastObjectType in ('PACKAGE', 'FUNCTION', 'PROCEDURE','SEQUENCE') then
         vRtnStr := vRtnStr || put_chars(' ', 3) || object_list_tab(i).object_name ||
                               put_chars(' ', 31 - length(object_list_tab(i).object_name)) || 
                               object_list_tab(i).status || 
                               put_chars(' ', 9 - length(object_list_tab(i).status)) ||
                               'Created: ' || to_char(object_list_tab(i).object_created, 'YYYY-MM-DD') || ', ' ||
                               'Last Updated: ' || to_char(object_list_tab(i).object_last_ddl_time, 'YYYY-MM-DD') ||
                               chr(10);
      end if;

   end loop;

   vRtnStr := vRtnStr || chr(10) || put_chars('-', 88) || chr(10) ||
              '= END OF REPORT: OBJECTS OWNED BY ' || vSchemaName || '@' || vDatabaseName || chr(10) ||
              put_chars('=', 88);

   return vRtnStr;

exception
   when excObjNotFound then
      return put_chars('=', nLineLength) || chr(10) ||
                 '= REPORT: OBJECTS OWNED BY ' || vSchemaName || '@' || vDatabaseName ||

                 put_chars('-', 88) || chr(10) ||
                 '= END OF REPORT: OBJECTS OWNED BY ' || vSchemaName || '@' || vDatabaseName || chr(10) ||
                 put_chars('=', 88);

   when others then
      return 'ERROR: ' || sqlerrm;

function ftGetModuleParamList           (pvDatabaseName                 IN     varchar2, 
                                         pvSchemaName                   IN     varchar2,
                                         pvPackageName                  IN     varchar2, 
                                         pvModuleName                   IN     varchar2 := null)
  return param_info_return_tab_type 
         pipelined is

   vDatabaseName                  varchar2(128) := upper(ltrim(rtrim(pvDatabaseName)));
   vSchemaName                    varchar2(128) := upper(ltrim(rtrim(pvSchemaName)));
   vPackageName                   varchar2(128) := upper(ltrim(rtrim(pvPackageName)));
   vModuleName                    varchar2(128) := upper(ltrim(rtrim(pvModuleName)));

   nFoo                           number;
   vStatement                     varchar2(4000);

   excObjNotFound                 exception;

begin

   vStatement := 'select da.object_name, da.position, da.argument_name, da.data_type, da.in_out, da.subprogram_id ' ||
                 '  from dba_arguments@' || vDatabaseName || ' da ' ||
                 ' where da.owner = ''' || vSchemaName || '''' ||
                 '   and (   (da.package_name = ''' || vPackageName || ''' and ''' || vPackageName || ''' is not null)' ||
                 '        or (''' || vPackageName || ''' is null and da.package_name is null and da.object_name = ''' || vModuleName || ''')' ||
                 '       )' ||
                 ' order by da.subprogram_id, da.position';

--dbms_output.put_line(vStatement);

   execute immediate vStatement
      bulk collect into param_info_tab;

   for i in 1 .. nvl(param_info_tab.last, 0) LOOP

      if param_info_tab(i).position <> 0 then

      pipe row(param_info_rec(param_info_tab(i).object_name,
                              param_info_tab(i).position,
                              param_info_tab(i).argument_name,
                              param_info_tab(i).data_type,
                              param_info_tab(i).in_out,
                              param_info_tab(i).subprogram_id));
      end if;

   end loop;

   return;

exception

   when excObjNotFound then
      pipe row (param_info_rec(null, null, 'ERROR - Object not found.', null, null, null));
      return;

   when others then
      pipe row (param_info_rec(null, null, 'ERROR - ' || sqlerrm, null, null, null));
      return;

end ftGetModuleParamList;


function ftGetPackageModuleList         (pvDatabaseName                 IN     varchar2, 
                                         pvSchemaName                   IN     varchar2, 
                                         pvPackageName                  IN     varchar2)
  return module_info_return_tab_type 
         pipelined is
   
   vDatabaseName                  varchar2(128)  := upper(ltrim(rtrim(pvDatabaseName)));
   vSchemaName                    varchar2(128)  := upper(ltrim(rtrim(pvSchemaName)));
   vPackageName                   varchar2(128)  := upper(ltrim(rtrim(pvPackageName)));

   nFoo                           number;
   vStatement                     varchar2(4000);
   vStatement2                    varchar2(4000);
   nNumFound                      number := 0;
   vProcOrFunc                    varchar2(20) := null;

   vStatement := 'select dp.procedure_name, ' || chr(10) ||
                 '       case when da.data_type is not null then ''FUNCTION'' else ''PROCEDURE'' end "MODULE_TYPE", ' || chr(10) ||
                 '       dp.subprogram_id, ' || chr(10) ||
                 '       case when da.data_type like ''%BOOLEAN%'' then ''BOOLEAN'' else da.data_type end "RETURN_TYPE" ' || chr(10) ||

function ftGetObjectList               (pvDatabaseName                 IN     varchar2,
                                        pvSchemaName                   IN     varchar2 DEFAULT NULL,
                                        pvObjectType                   IN     varchar2 DEFAULT NULL)
  return object_list_return_tab_type 
         pipelined is

   vDatabaseName                  varchar2(128)   := upper(ltrim(rtrim(pvDatabaseName)));
   vSchemaName                    varchar2(128);
   vObjectType                    varchar2(128);

   vStatement1                    varchar2(4000);
   vStatement2                    varchar2(4000);
   vStatement3                    varchar2(4000);
   vLastObjectType                varchar2(128)  := 'zzz';
   vFirstRow                      varchar2(1)    := 'Y';
   vReturnType                    varchar2(128);
   dDate1                         date;
   dDate2                         date;

   rObjectList                    object_list_rec;

   excObjNotFound                 exception;

begin
-- dbms_output.put_line('DEBUG 1');

   if pvSchemaName is null then
      vSchemaName := 'ALL';
   else
      vSchemaName := upper(ltrim(rtrim(pvSchemaName)));
   end if;
-- dbms_output.put_line('DEBUG 2');

   if pvObjectType is null then
      vObjectType := 'ALL';
   else
      vObjectType := upper(ltrim(rtrim(pvObjectType)));
   end if;
-- dbms_output.put_line('DEBUG 3');

   vStatement1 := 'select do.object_type, do.object_name, do.status, ' || chr(10) ||
                  '       do.created "OBJECT_CREATED", ' || chr(10) ||
                  '       do.last_ddl_time "OBJECT_LAST_DDL_TIME", ' || chr(10) ||
                  '       null "BODY_CREATED", null "BODY_LAST_DDL_TIME", null "RETURN_TYPE" ' || chr(10) ||
                  '  from dba_objects@' || vDatabaseName || ' do ' || chr(10) ||
                  ' where (instr(''' || vObjectType || ''', do.object_type) > 0 or ''' || vObjectType || ''' = ''ALL'')' || chr(10) ||
                  '   and (''' || vSchemaName || ''' = ''ALL'' or do.owner = ''' || vSchemaName || ''')' || chr(10) ||
                  '   and do.object_type in (''TABLE'', ''VIEW'', ''PACKAGE'', ''FUNCTION'', ''PROCEDURE'',''SEQUENCE'')' || chr(10) ||
                  '   and instr(do.object_name, ''$'') = 0 ' || chr(10) ||
                  ' order by decode(do.object_type, ''TABLE'', 1, ''VIEW'', 2, ''PACKAGE'', 3, ' || chr(10) ||
                  '                                 ''PROCEDURE'', 4, ''FUNCTION'', 5, ''SEQUENCE'', 6), do.object_name';

--dbms_output.put_line(vStatement1);

   execute immediate vStatement1
      bulk collect into object_list_tab;
-- dbms_output.put_line('DEBUG 5');

   if nvl(object_list_tab.last, 0) = 0 then
      raise excObjNotFound;
   end if;

   for i in 1 .. nvl(object_list_tab.last, 0) LOOP

      if object_list_tab(i).object_type = 'FUNCTION' then

         vStatement2 := 'select data_type ' || chr(10) ||
                        '  from dba_arguments@' || vDatabaseName || ' da ' || chr(10) ||
                        ' where da.owner = ''' || vSchemaName || '''' || chr(10) ||
                        '   and da.object_name = ''' || object_list_tab(i).object_name || '''' || chr(10) ||
                        '   and da.package_name is null ' || chr(10) ||
                        '   and da.position = 0 ';

         begin

            vReturnType := null;

            execute immediate vStatement2
               bulk collect into object_data_type_tab; -- nNumFound;

            vReturnType := object_data_type_tab(1).data_type;

         exception 
            when no_data_found then
               dbms_output.put_line('... WARNING: FUNCTION ' || object_list_tab(i).object_name || ' has no RETURN TYPE.');
               dDate1 := null;
               dDate2 := null;

            when others then
               dbms_output.put_line('... ERROR: ' || sqlerrm || ' for Package ' || object_list_tab(i).object_name || '.');
         end;

      else
         vReturnType := null;
      end if;
      if object_list_tab(i).object_type = 'PACKAGE' then

         vStatement3 := 'select do.created, do.last_ddl_time ' || chr(10) ||
                        '  from dba_objects@' || vDatabaseName || ' do ' || chr(10) ||
                        ' where do.object_type = ''PACKAGE BODY'' ' || chr(10) ||
                        '   and do.owner = ''' || vSchemaName || ''' ' || chr(10) ||
                        '   and do.object_name = ''' || object_list_tab(i).object_name || ''' ';

         begin
            execute immediate vStatement3
               bulk collect into object_date_tab;

            dDate1 := object_date_tab(1).created;
            dDate2 := object_date_tab(1).last_ddl_time;

         exception
            when no_data_found then
               dbms_output.put_line('... WARNING: PACKAGE ' || object_list_tab(i).object_name || ' has no PACKAGE BODY.');
               dDate1 := null;
               dDate2 := null;

            when others then
               dbms_output.put_line('... ERROR: ' || sqlerrm || ' for Package ' || object_list_tab(i).object_name || '.');
         end;

      else
         dDate1 := null;
         dDate2 := null;
         vStatement2 := null;
      end if;

      pipe row (object_list_rec(object_list_tab(i).object_type,
                                object_list_tab(i).object_name, 
                                object_list_tab(i).status,
                                object_list_tab(i).object_created,
                                object_list_tab(i).object_last_ddl_time,
                                dDate1, --object_date_tab(1).created,
                                dDate2, -- object_date_tab(1).last_ddl_time,
                                vReturnType));
   end loop;

exception
   when excObjNotFound then
--      pipe row (object_list_rec('N/A', 'ERROR: Object not found.', null, null, null, null, null, null));
      return;
   when others then
      dbms_output.put_line('------------------------------------------');
      dbms_output.put_line('Query1:' || chr(10) || vStatement1);
      dbms_output.put_line('------------------------------------------');
      dbms_output.put_line('Query2:' || chr(10) || vStatement2);
      dbms_output.put_line('------------------------------------------');
      dbms_output.put_line('Query3:' || chr(10) || vStatement3);
      pipe row (object_list_rec('N/A', 'ERROR: ' || sqlerrm || ' in ftGetObjectList.', null, null, null, null, null, null));
      return;

end ftGetObjectList;

function  fbValidObjectType             (pvObjectType                   IN     varchar2)
  return  boolean is

   bRtnVal                     boolean := FALSE;

begin

   if nvl(upper(pvObjectType), '-') in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION','SEQUENCE') then
      bRtnVal := TRUE;
   else
      bRtnVal := FALSE;
   end if;

   return bRtnVal;

end fbValidObjectType;

function  fbValidEnvironmentName        (pvEnvName                      IN     varchar2)
  return  boolean is

   nCount                      number  := 0;
   bRtnVal                     boolean := FALSE;

begin

   select count(*)
     into nCount
     from da_environments e
    where upper(pvEnvName) = e.environment_nm;

   if nCount = 1 then
      bRtnVal := TRUE;
   else
      bRtnVal := FALSE;
   end if;

   return bRtnVal;

end fbValidEnvironmentName;

function  fbValidDatabaseName        (pvDatabaseName                   IN     varchar2)
  return  boolean is

   nCount                      number  := 0;
   bRtnVal                     boolean := FALSE;

begin
   select count(*)
     into nCount
     from da_databases d
    where upper(pvDatabaseName) = d.database_nm;

   if nCount = 1 then
      bRtnVal := TRUE;
   else
      bRtnval := FALSE;
   end if;

   return bRtnVal;

end fbValidDatabaseName;

function  fbValidSchemaName        (pvSchemaName                      IN     varchar2)
  return  boolean is

   nCount                      number  := 0;
   bRtnVal                     boolean := FALSE;

begin

   select count(*)
     into nCount
     from da_schemas s
    where upper(pvSchemaName) = s.schema_nm;

   if nCount = 1 then
      bRtnVal := TRUE;
   else
      bRtnVal := FALSE;
   end if;

   return bRtnVal;

end fbValidSchemaName;


procedure prAddSingleDescriptSession    (pvDatabaseName                 IN     varchar2,
                                         pvSchemaName                   IN     varchar2,
                                         pvObjectType                   IN     varchar2, 
                                         pvDebugYN                      IN     varchar2 := 'N')
       is

-- updates:
--   ALL records in da_description sessions for this combination of specific DB, schema, and object_type
-- creates:
--   1   da_description_session record
--   N   da_described_objects records
--   N*M da_described_table_columns records

   vDatabaseName                  varchar2(100) := upper(ltrim(rtrim(pvDatabaseName)));
   vSchemaName                    varchar2(100) := upper(ltrim(rtrim(pvSchemaName)));
   vObjectType                    varchar2(100) := upper(ltrim(rtrim(pvObjectType)));

   vObjectTypeMixedCase           varchar2(128);
   nObjectsAdded                  number        := 0;
   dDescribedDtm                  date          := sysdate;
   dBodyCreatedDtm                date;
   dBodyLastDdlDtm                date;
   nItemsFound                    number;

   nDescriptionSessionId          number;
   nDescribedObjectId             number;
   nDescribedTableColumnId        number;
   nDescribedModuleId             number;
   nDescribedModuleParamId        number;
   nDescribedObjectGrantId        number;

   bSkipObject                    boolean;

   nGrantsFound                   number        := 0;
   nGrantsNotFound                number        := 0;
   nGrantsErrorCount              number        := 0;
   nSynonymsFound                 number        := 0;
   nSynonymsNotFound              number        := 0;
   nSynonymsErrorCount            number        := 0;

   vErrorStr                      varchar2(200) := '';
   vErrorSep                      varchar2(2)   := '';
   
   excNoObjectTypeFound           exception;
   excIncompleteParams            exception;
   excBadParamValue               exception;

begin

   if vDatabaseName is null or vSchemaName is null or vObjectType is null then

      if vDatabaseName is null then
         vErrorStr := vErrorStr || vErrorSep || 'DATABASE';
         vErrorSep := ', ';
      end if;
      if vSchemaName is null then
         vErrorStr := vErrorStr || vErrorSep || 'SCHEMA';
         vErrorSep := ', ';
      end if;
      if vObjectType is null then
         vErrorStr := vErrorStr || vErrorSep || 'OBJECT_TYPE';
         vErrorSep := ', ';
      end if;
     
      raise excIncompleteParams;
   end if;

   begin
      select object_type_mixed_case_nm 
        into vObjectTypeMixedCase
        from da_object_types
       where object_type = vObjectType;
   exception
      when no_data_found then
         raise excnoObjectTypeFound;
   end;

   if not (fbValidDatabaseName(vDatabaseName) and fbValidSchemaName(vSchemaName) and fbValidObjectType(vObjectType)) then

      if not fbValidDatabaseName(vDatabaseName) then
         vErrorStr := vErrorSep || 'DATABASE_NAME: ' || vDatabaseName;
         vErrorSep := ', ';
      end if;
      if not fbValidSchemaName(vSchemaName) then
         vErrorStr := vErrorSep || 'SCHEMA_NAME: ' || vSchemaName;
         vErrorSep := ', ';
      end if;
      if not fbValidObjectType(vObjectType) then
         vErrorStr := vErrorSep || 'OBJECT_TYPE: ' || vObjectType;
         vErrorSep := ', ';
      end if;
      raise excBadParamValue;
   end if;

   dbms_output.put_line(vDivider1);
   dbms_output.put_line('DATA EXTRACT for ' || vObjectTypeMixedCase || 's in ' || vSchemaName || '@' || vDatabaseName || ', starting at ' 
                           || to_char(sysdate, vTimeFmt));
   dbms_output.put_line(vDivider1);
   dbms_output.put_line(vDivider2);


   select dese_seq.nextval into nDescriptionSessionId from dual;

   update da_description_sessions
      set current_version_yn = 'N'
    where current_version_yn = 'Y'
      and database_nm = vDatabaseName
      and schema_nm = vSchemaName
      and object_type = vObjectType;

   insert into da_description_sessions
        ( DESCRIPTION_SESSION_ID        ,    --     NOT NULL  NUMBER(, )
          DATABASE_NM                   ,    --     NOT NULL  VARCHAR2(128)
          SCHEMA_NM                     ,    --     NOT NULL  VARCHAR2(128)
          OBJECT_TYPE                   ,    --     NOT NULL  VARCHAR2(128)
          SESSION_START_DTM             ,    --  05 NOT NULL  DATE
          OBJECT_COUNT                  ,    --     NOT NULL  NUMBER(, )
          CURRENT_VERSION_YN            ,    --     NOT NULL  VARCHAR2(1)
          SESSION_END_DTM                    --               DATE
        ) values (
          nDescriptionSessionId,
          vDatabaseName,
          vSchemaName,
          vObjectType,
          sysdate, --  05
          0,
          'Y',
          null
        );

   commit;

   for obj in (select object_type, object_name, return_type,
                      status, object_created, object_last_ddl_time, body_created, body_last_ddl_time
                 from table(pkg_da_describe_kw.ftGetObjectList(vDatabaseName, 
                                                            vSchemaName, 
                                                            vObjectType))
              ) loop

      bSkipObject := FALSE;

      for exc in (select exclusion_txt, prefix_yn, suffix_yn, middle_yn
                    from da_exclusion_texts
                   where object_type = vObjectType
                   order by 1
               ) loop

         if exc.prefix_yn = 'Y' then
            if obj.object_name like exc.exclusion_txt || '%' then
               bSkipObject := TRUE;
            end if;
         elsif exc.suffix_yn = 'Y' then
            if obj.object_name like '%' || exc.exclusion_txt then
               bSkipObject := TRUE;
            end if;
         elsif exc.middle_yn = 'Y' then
            if obj.object_name like '%' || exc.exclusion_txt || '%' then
               bSkipObject := TRUE;
            end if;
         end if;
      end loop;

      if bSkipObject then
         dbms_output.put_line('... Skipping: ' || obj.object_name);
      else

         select deob_seq.nextval into nDescribedObjectId from dual;
   
         insert into da_described_objects
              ( DESCRIBED_OBJECT_ID           ,    --     NOT NULL  NUMBER(, )
                DESCRIPTION_SESSION_ID        ,    --     NOT NULL  NUMBER(, )
                OBJECT_NM                     ,    --     NOT NULL  VARCHAR2(128)
                CHILD_ITEM_COUNT              ,    --     NOT NULL  NUMBER(, )
                OBJECT_CREATED_DTM            ,    --               DATE
                OBJECT_LAST_DDL_DTM           
              ) values (
                nDescribedObjectId,
                nDescriptionSessionId,
                obj.object_name,
                0,
                obj.object_created,
                obj.object_last_ddl_time
              );

         nObjectsAdded := nObjectsAdded + 1;

         nItemsFound := 0;

         if vObjectType = 'TABLE' then

            for cols in (select column_id, column_name, 
                                data_type, data_length, 
                                data_precision, data_scale, nullable_yn, data_default
                           from table(pkg_da_describe_kw.ftGetTableColumnList(vDatabaseName, 
                                                                           vSchemaName, 
                                                                           obj.object_name))
                        ) loop

               select detc_seq.nextval into nDescribedTableColumnId from dual;

               insert into da_described_table_columns
                    ( DESCRIBED_TABLE_COLUMN_ID     ,    --     NOT NULL  NUMBER(, )
                      DESCRIBED_OBJECT_ID           ,    --               NUMBER(, )
                      COLUMN_NM                     ,    --               VARCHAR2(128)
                      COLUMN_ID                     ,    --               NUMBER(, )
                      DATA_TYPE                     ,    --  05           VARCHAR2(128)
                      DATA_LENGTH                   ,    --               NUMBER(, )
                      DATA_PRECISION                ,    --               NUMBER(, )
                      DATA_SCALE                    ,    --               NUMBER(, )
                      NULLABLE_YN                   ,     --               VARCHAR2(1)
		      DATA_DEFAULT
                    ) values (
                      nDescribedTableColumnId,
                      nDescribedObjectId,
                      cols.column_name,
                      cols.column_id,
                      cols.data_type,
                      cols.data_length,
                      cols.data_precision,
                      cols.data_scale,
                      cols.nullable_yn,
		      cols.data_default
                    );


               nItemsFound := nItemsFound + 1;

            end loop;

            update da_described_objects
               set child_item_count = nItemsFound
             where described_object_id = nDescribedObjectId;

            commit;

         elsif vObjectType = 'VIEW' then

            for cols in (select column_id, column_name, 
                                data_type, data_length, 
                                data_precision, data_scale, nullable_yn, data_default
                           from table(pkg_da_describe_kw.ftGetTableColumnList(vDatabaseName, 
                                                                           vSchemaName, 
                                                                           obj.object_name))
                        ) loop

               select detc_seq.nextval into nDescribedTableColumnId from dual;

               insert into da_described_table_columns
                    ( DESCRIBED_TABLE_COLUMN_ID     ,    --     NOT NULL  NUMBER(, )
                      DESCRIBED_OBJECT_ID           ,    --               NUMBER(, )
                      COLUMN_NM                     ,    --               VARCHAR2(128)
                      COLUMN_ID                     ,    --               NUMBER(, )
                      DATA_TYPE                     ,    --  05           VARCHAR2(128)
                      DATA_LENGTH                   ,    --               NUMBER(, )
                      DATA_PRECISION                ,    --               NUMBER(, )
                      DATA_SCALE                    ,    --               NUMBER(, )
                      NULLABLE_YN                   ,     --               VARCHAR2(1)
		      DATA_DEFAULT
                    ) values (
                      nDescribedTableColumnId,
                      nDescribedObjectId,
                      cols.column_name,
                      cols.column_id,
                      cols.data_type,
                      cols.data_length,
                      cols.data_precision,
                      cols.data_scale,
                      cols.nullable_yn,
		      cols.data_default
                    );


               nItemsFound := nItemsFound + 1;

            end loop;

            update da_described_objects
               set child_item_count = nItemsFound
             where described_object_id = nDescribedObjectId;

            commit;

         elsif vObjectType = 'PACKAGE' then

            declare
               type num_tab_type is table of number index by binary_integer;
               tModuleIds num_tab_type;
            begin

               for mods in (select *
                              from table(pkg_da_describe_kw.ftGetPackageModuleList(vDatabaseName, 
                                                                                vSchemaName, 
                                                                                obj.object_name))
                           ) loop

                  select demo_seq.nextval into nDescribedModuleId from dual;

                  tModuleIds(mods.subprogram_id) := nDescribedModuleId;

                  insert into da_described_modules
                       ( DESCRIBED_MODULE_ID           ,    --     NOT NULL  NUMBER(, )
                         DESCRIBED_OBJECT_ID           ,    --               NUMBER(, )
                         MODULE_NM                     ,    --               VARCHAR2(128)
                         MODULE_ORDER_NO               ,    --               NUMBER(, )
                         MODULE_TYPE_DSC               ,
                         MODULE_RETURN_VALUE_TYPE_DSC
                       ) values (
                         nDescribedModuleId,
                         nDescribedObjectId,
                         mods.module_name,
                         mods.subprogram_id,
                         mods.module_type,
                         mods.return_type);

                  nItemsFound := nItemsFound + 1;

               end loop;

               for params in (select *
                               from table(pkg_da_describe_kw.ftGetModuleParamList(vDatabaseName, 
                                                                               vSchemaName, 
                                                                               obj.object_name))
                            ) loop

                  if tModuleIds.exists(params.subprogram_id) then
                     insert into da_described_module_params
                          ( DESCRIBED_MODULE_PARAM_ID     ,    --     NOT NULL  NUMBER(, )
                            DESCRIBED_MODULE_ID           ,    --     NOT NULL  NUMBER(, )
                            PARAM_NM                      ,    --     NOT NULL  VARCHAR2(128)
                            PARAM_ORDER_NO                ,    --     NOT NULL  NUMBER(, )
                            PARAM_TYPE                    ,    --  05 NOT NULL  VARCHAR2(200)
                            PARAM_DIRECTION                    --     NOT NULL  VARCHAR2(3)
                          ) values (
                            demp_seq.nextval,
                            tModuleIds(params.subprogram_id),
                            params.argument_name,
                            params.position,
                            params.data_type,
                            params.in_out);
                  end if;

               end loop;

            end;

            update da_described_objects
               set child_item_count = nItemsFound,
                   body_created_dtm = obj.body_created,
                   body_last_ddl_dtm = obj.body_last_ddl_time
             where described_object_id = nDescribedObjectId;

         elsif vObjectType in ('PROCEDURE', 'FUNCTION') then

            select demo_seq.nextval into nDescribedModuleId from dual;

            insert into da_described_modules
                 ( DESCRIBED_MODULE_ID           ,    --     NOT NULL  NUMBER(, )
                   DESCRIBED_OBJECT_ID           ,    --               NUMBER(, )
                   MODULE_NM                     ,    --               VARCHAR2(128)
                   MODULE_ORDER_NO               ,    --               NUMBER(, )
                   MODULE_TYPE_DSC               ,
                   MODULE_RETURN_VALUE_TYPE_DSC
                 ) values (
                   nDescribedModuleId,
                   nDescribedObjectId,
                   obj.object_name,
                   null,
                   vObjectType,
                   obj.return_type);

            nItemsFound := nItemsFound + 1;

            for params in (select *
                             from table(pkg_da_describe_kw.ftGetModuleParamList(vDatabaseName, 
                                                                             vSchemaName, 
                                                                             null, 
                                                                             obj.object_name))
                           ) loop

               insert into da_described_module_params
                    ( DESCRIBED_MODULE_PARAM_ID     ,    --     NOT NULL  NUMBER(, )
                      DESCRIBED_MODULE_ID           ,    --     NOT NULL  NUMBER(, )
                      PARAM_NM                      ,    --     NOT NULL  VARCHAR2(128)
                      PARAM_ORDER_NO                ,    --     NOT NULL  NUMBER(, )
                      PARAM_TYPE                    ,    --  05 NOT NULL  VARCHAR2(200)
                      PARAM_DIRECTION                    --     NOT NULL  VARCHAR2(3)
                    ) values (
                      demp_seq.nextval,
                      nDescribedModuleId,
                      params.argument_name,
                      params.position,
                      params.data_type,
                      params.in_out);

            end loop;

	 elsif vObjectType = 'SEQUENCE' then

            null; --- nothing to do aside from grants (handled below)

         end if;

         -- process Grants for all specified object types
         if vObjectType in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION', 'SEQUENCE') then

            for grants in (select grantor, grantee, privilege, grantable, inherited
                             from table(ftGetObjectGrantList(vDatabaseName, 
                                                             vSchemaName, 
                                                             obj.object_name))
                          ) loop

               if (grants.grantor) <> 'No Grants Found.' and instr(grants.grantor, 'ERROR -') = 0 then

                  select deog_seq.nextval into nDescribedObjectGrantId from dual;

                  insert into da_described_object_grants
                       ( DESCRIBED_OBJECT_GRANT_ID     ,     --     NOT NULL  NUMBER(, )
                         DESCRIBED_OBJECT_ID           ,    --     NOT NULL  NUMBER(, )
                         GRANTOR                       ,    --     NOT NULL  VARCHAR2(128)
                         GRANTEE                       ,    --     NOT NULL  VARCHAR2(128)
                         PRIVILEGE                     ,    --     NOT NULL  VARCHAR2(40)
                         GRANTABLE                     ,    --     NOT NULL  VARCHAR2(3)
                         INHERITED                          --     NOT NULL  VARCHAR2(3)
                       ) values (
                         nDescribedObjectGrantId,
                         nDescribedObjectId,
                         grants.grantor,
                         grants.grantee,
                         grants.privilege,
                         grants.grantable,
                         grants.inherited);

                  nGrantsFound := nGrantsFound + 1;
               elsif (grants.grantor) = 'No Grants Found.' then
                  nGrantsNotFound := nGrantsNotFound + 1;
               elsif instr(grants.grantor, 'ERROR -') > 0 then
                  dbms_output.put_line('Grant extraction error for ' || obj.object_name || ': ' || grants.grantor);
                  nGrantsErrorCount := nGrantsErrorCount + 1;
               end if;

            end loop;

         end if;

         -- process Synonyms for all specified object types
         if vObjectType in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION', 'SEQUENCE') then

            for syns in (select synonym_name, table_owner, db_link
                             from table(ftGetObjectSynonymList(vDatabaseName, 
                                                               vSchemaName, 
                                                               obj.object_name))
                          ) loop

               if (syns.synonym_name) <> 'No Synonyms Found.' and instr(syns.synonym_name, 'ERROR -') = 0 then

                  insert into da_described_object_synonyms
                       ( DESCRIBED_OBJECT_SYNONYM_ID   ,    --     NOT NULL  NUMBER(, )
                         DESCRIBED_OBJECT_ID           ,    --     NOT NULL  NUMBER(, )
                         SYNONYM_NM                    ,    --     NOT NULL  VARCHAR2(128)
                         OBJECT_OWNER                  ,    --               VARCHAR2(128)
                         DB_LINK                            --               VARCHAR2(128)
                       ) values (
                         deos_seq.nextval,
                         nDescribedObjectId,
                         syns.synonym_name,
                         syns.table_owner,
                         syns.db_link);
                         
                  nSynonymsFound := nSynonymsFound + 1;

               elsif (syns.synonym_name) = 'No Synonyms Found.' then
                  nSynonymsNotFound := nSynonymsNotFound + 1;
               elsif instr(syns.synonym_name, 'ERROR -') > 0 then
                  dbms_output.put_line('Synonym extraction error for ' || obj.object_name || ': ' || syns.synonym_name);
                  nSynonymsErrorCount := nSynonymsErrorCount + 1;
               end if;

            end loop;

         end if;

      end if;

   end loop;

   update da_description_sessions
      set session_end_dtm = sysdate
    where description_session_id = nDescriptionSessionId;

   if nObjectsAdded = 0 then
      dbms_output.put_line('COMPLETE - Zero ' || vObjectTypeMixedCase || 's were found.');
   else
      dbms_output.put_line('COMPLETE - Added ' || nObjectsAdded || ' ' || vObjectTypeMixedCase || 's.');
      dbms_output.put_line('... ' || to_char(nGrantsFound, '999,999') || ' grants were found for these ' || 
                              vObjectTypeMixedCase || 's, and');
      dbms_output.put_line('... ' || to_char(nGrantsNotFound, '999,999') || ' ' || vObjectTypeMixedCase || 
                              's had no grants.');
      if nGrantsErrorCount > 0 then
         dbms_output.put_line('... ' || to_char(nGrantsErrorCount, '999,999') || ' ' || vObjectTypeMixedCase || 
                              's threw extraction errors (check output).');
      end if;
      dbms_output.put_line('... ' || to_char(nSynonymsFound, '999,999') || ' synonyms were found pointing to these ' || 
                              vObjectTypeMixedCase || 's, and');
      dbms_output.put_line('... ' || to_char(nSynonymsNotFound, '999,999') || ' ' || vObjectTypeMixedCase || 
                              's had no synonyms.');
      if nSynonymsErrorCount > 0 then
         dbms_output.put_line('... ' || to_char(nSynonymsErrorCount, '999,999') || ' ' || vObjectTypeMixedCase || 
                              's threw extraction errors (check output).');
      end if;
   end if;

   dbms_output.put_line(vDivider2);
   dbms_output.put_line(vDivider1);
   dbms_output.put_line('DATA EXTRACT for ' || vObjectTypeMixedCase || 's in ' ||  vSchemaName || '@' || 
                           vDatabaseName || ', completed at ' || to_char(sysdate, vTimeFmt));
   dbms_output.put_line(vDivider1);

   commit;

exception

   when excIncompleteParams then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' was/were not specified.');

   when excBadParamValue then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' is/are not valid.');

   when others then
      dbms_output.put_line('ERROR - ' || sqlerrm || '.');

end prAddSingleDescriptSession;


procedure prAddMultipleDescriptSessions (pvEnvName                      IN     varchar2 := '',
                                         pvDatabaseName                 IN     varchar2 := '',
                                         pvSchemaName                   IN     varchar2 := '',
                                         pvObjectType                   IN     varchar2 := '')
       is

   vEnvName                       varchar2(100) := upper(ltrim(rtrim(pvEnvName)));           -- null if doing all environments
   vDatabaseName                  varchar2(100) := upper(ltrim(rtrim(pvDatabaseName)));      -- null if doing all databases
   vSchemaName                    varchar2(100) := upper(ltrim(rtrim(pvSchemaName)));        -- null if doing all schemae
   vObjectType                    varchar2(100) := upper(ltrim(rtrim(pvObjectType)));        -- null if doing all object types

   nCount                         number        := 0;
   vErrorStr                      varchar2(200) := '';
   vErrorSep                      varchar2(2)   := '';
   
   excIncompleteParams            exception;
   excBadParamValue               exception;

begin

   if (   vEnvName is not null       and not fbValidEnvironmentName(vEnvName)
       or vDatabaseName is not null  and not fbValidDatabaseName(vDatabaseName)
       or vSchemaName is not null    and not fbValidSchemaName(vSchemaName)
       or vObjectType is not null    and not fbValidObjectType(vObjectType)
      ) then

      if vEnvName is not null and not fbValidEnvironmentName(vEnvName) then
         vErrorStr := vErrorStr || vErrorSep || 'ENVIRONMENT_NAME: ' || vEnvName;
         vErrorSep := ', ';
      end if;
      if vDatabaseName is not null and not fbValidDatabaseName(vDatabaseName) then
         vErrorStr := vErrorStr || vErrorSep || 'DATABASE_NAME: ' || vDatabaseName;
         vErrorSep := ', ';
      end if;
      if vSchemaName is not null and not fbValidSchemaName(vSchemaName) then
         vErrorStr := vErrorStr || vErrorSep || 'SCHEMA_NAME: ' || vSchemaName;
         vErrorSep := ', ';
      end if;
      if vObjectType is not null and not fbValidObjectType(vObjectType) then
         vErrorStr := vErrorStr || vErrorSep || 'OBJECT_TYPE: ' || vObjectType;
         vErrorSep := ', ';
      end if;
      raise excBadParamValue;
   end if;

   for objType in (select object_type
                     from da_object_types objt
                    where objt.active_use_yn = 'Y'
                      and nvl(vObjectType, objt.object_type) = objt.object_type
                    order by objt.sort_order_no
                  ) loop

      for location in (select a.application_nm, e.environment_nm, d.database_nm, sdi.schema_nm, 
                              sdi.schema_nm || '@' || sdi.database_nm "DATABASE_LOCATION"
                         from da_schema_database_instances sdi,
                              da_schemas s,
                              da_applications a,
                              da_environments e,
                              da_databases d
                        where sdi.database_nm = d.database_nm
                          and nvl(vDatabaseName, d.database_nm) = d.database_nm
                          and d.active_use_yn = 'Y'
                          and sdi.environment_nm = e.environment_nm
                          and nvl(vEnvName, e.environment_nm) = e.environment_nm
                          and e.active_use_yn = 'Y'
                          and sdi.schema_nm = s.schema_nm
                          and s.application_nm = a.application_nm
                          and nvl(vSchemaName, s.schema_nm) = s.schema_nm
                          and a.active_use_yn = 'Y'
                          and s.active_use_yn = 'Y'
                        order by 1, e.sort_order_no, decode(sdi.database_nm, 'TRNJ', 100, 'TRNE', 200, 1), 3
                      ) loop

         pkg_da_describe_kw.prAddSingleDescriptSession(location.database_nm, location.schema_nm, objType.object_type);

      end loop;
   end loop;

exception

   when excIncompleteParams then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' was/were not specified.');

   when excBadParamValue then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' is/are not valid.');

   when others then
      dbms_output.put_line('ERROR: ' || sqlerrm || '.');

end prAddMultipleDescriptSessions;

procedure prCompareEnvironments         (pvDatabase1                    IN     varchar2,
                                         pvDatabase2                    IN     varchar2,
                                         pvSchemaName                   IN     varchar2,
                                         pvObjectType                   IN     varchar2)
       is

   vDatabase1                     varchar2(100) := upper(ltrim(rtrim(pvDatabase1)));
   vDatabase2                     varchar2(100) := upper(ltrim(rtrim(pvDatabase2)));
   vSchemaName                    varchar2(100) := upper(ltrim(rtrim(pvSchemaName)));
   vObjectType                    varchar2(100) := upper(ltrim(rtrim(pvObjectType)));

   nFoo                           number;
   tObjectName                    varchar2(30);
   tSubObjectName                 varchar2(30);

   nDifference1                   number := 0;
   nDifference2                   number := 0;
   nDifference3                   number := 0;
   nDifference4                   number := 0;
   nDifference5                   number := 0;
   vObjectTypeMixedCase           varchar2(128);
   vPadStr1                       varchar2(200);
   vPadStr2                       varchar2(200);
   vPadStr3                       varchar2(200);
   vLastPrintedObject             varchar2(128)  := 'zzz';
   nSectionNo                     number := 0;
   tOutputSpacer                  varchar2(20)   := '';

   vErrorStr                      varchar2(200)  := '';
   vErrorSep                      varchar2(2)    := '';

   excNoObjectTypeFound           exception;
   excIncompleteParams            exception;

begin

   if vDatabase1 is null or vDatabase2 is null or vSchemaName is null or vObjectType is null then

      if vDatabase1 is null then
         vErrorStr := vErrorSep || 'DATABASE1';
         vErrorSep := ', ';
      end if;
      if vDatabase2 is null then
         vErrorStr := vErrorStr || vErrorSep || 'DATABASE2';
         vErrorSep := ', ';
      end if;
      if vSchemaName is null then
         vErrorStr := vErrorStr || vErrorSep || 'SCHEMA';
         vErrorSep := ', ';
      end if;
      if vObjectType is null then
         vErrorStr := vErrorStr || vErrorSep || 'OBJECT_TYPE';
         vErrorSep := ', ';
      end if;
     
      raise excIncompleteParams;
   end if;

   begin
      select object_type_mixed_case_nm 
        into vObjectTypeMixedCase
        from da_object_types
       where object_type = vObjectType;
   exception
      when no_data_found then
         raise excnoObjectTypeFound;
   end;

   if vObjectType = 'TABLE' then
      tSubObjectName := 'column';
      tOutputSpacer  := '    ';
   elsif vObjectType = 'VIEW' then
      tSubObjectName := 'column';
      tOutputSpacer  := '     ';
   elsif vObjectType = 'PACKAGE' then
      tSubObjectName := 'module';
      tOutputSpacer  := '  ';
   elsif vObjectType = 'PROCEDURE' then
      tSubObjectName := 'parameter';
      tOutputSpacer  := '';
   elsif vObjectType = 'FUNCTION' then
      tSubObjectName := 'parameter';
      tOutputSpacer  := ' ';
   end if;

   dbms_output.put_line(vDivider1);
   dbms_output.put_line('COMPARING ' || vObjectTypeMixedCase || 's in ' || vSchemaName || 
                           '@' || vDatabase1 || ' vs ' || vSchemaName || '@' || vDatabase2);
   dbms_output.put_line(vDivider1);
   dbms_output.put_line(vDivider2);

   if vObjectType in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION', 'SEQUENCE') then

      nSectionNo := nSectionNo + 1;
      dbms_output.put_line(to_char(nSectionNo, 'RN') || ') ' || vObjectTypeMixedCase || 's in ' || vSchemaName || '@' || 
                              vDatabase1 || ' but not in ' || vSchemaName || '@' || vDatabase2);


      for x in (select deob.object_nm
                  from da_described_objects deob,
                       da_description_sessions dese
                 where dese.database_nm = vDatabase1
                   and dese.schema_nm = vSchemaName
                   and dese.object_type = vObjectType
                   and dese.current_version_yn = 'Y'
                   and deob.description_session_id = dese.description_session_id
                 order by object_nm
               ) loop

         begin
            select deob.object_nm
              into tObjectName
              from da_described_objects deob,
                   da_description_sessions dese
             where dese.database_nm = vDatabase2
               and dese.schema_nm = vSchemaName
               and dese.object_type = vObjectType
               and deob.object_nm = x.object_nm
               and dese.current_version_yn = 'Y'
               and deob.description_session_id = dese.description_session_id
               and rownum = 1;
         exception
            when no_data_found then
               dbms_output.put_line('... FOUND: ' || vObjectTypeMixedCase || ' ' || x.object_nm);
               nDifference1 := nDifference1 + 1;
            when others then
               dbms_output.put_line('ERROR1: ' || x.object_nm || ' - ' || sqlerrm);
               raise;
         end;
      end loop;

      if nDifference1 = 0 then
         dbms_output.put_line('... None found.');
      end if;

      dbms_output.put_line(vDivider2);

      nSectionNo := nSectionNo + 1;
      dbms_output.put_line(to_char(nSectionNo, 'RN') || ') ' || vObjectTypeMixedCase || 's in ' || vSchemaName || '@' || 
                              vDatabase2 || ' but not in ' || vSchemaName || '@' || vDatabase1);

      for x in (select deob.object_nm
                  from da_described_objects deob,
                       da_description_sessions dese
                 where dese.database_nm = vDatabase2
                   and dese.schema_nm = vSchemaName
                   and dese.object_type = vObjectType
                   and dese.current_version_yn = 'Y'
                   and deob.description_session_id = dese.description_session_id
                 order by deob.object_nm
               ) loop

         begin
            select deob.object_nm
              into tObjectName
              from da_described_objects deob,
                   da_description_sessions dese
             where dese.database_nm = vDatabase1
               and dese.schema_nm = vSchemaName
               and dese.object_type = vObjectType
               and deob.object_nm = x.object_nm
               and dese.current_version_yn = 'Y'
               and deob.description_session_id = dese.description_session_id
               and rownum = 1;

         exception
            when no_data_found then
               dbms_output.put_line('... FOUND: ' || vObjectTypeMixedCase || ' ' || x.object_nm);
               nDifference2 := nDifference2 + 1;
            when others then
               dbms_output.put_line('ERROR2: ' || sqlerrm);
               raise;
         end;
      end loop;

      if nDifference2 = 0 then
         dbms_output.put_line('... None found.');
      end if;

      dbms_output.put_line(vDivider2);

   end if;

   if vObjectType in ('TABLE', 'VIEW', 'PACKAGE') then

      nSectionNo := nSectionNo + 1;
      dbms_output.put_line(to_char(nSectionNo, 'RN') || ') ' || vObjectTypeMixedCase || 's in ' || vSchemaName || '@' || 
                              vDatabase1 || ' with differing ' || tSubObjectName || ' counts than in ' || vDatabase2);

      for x in (select deob1.object_nm, deob1.child_item_count "COUNT1", deob2.child_item_count "COUNT2"
                  from da_described_objects deob1,
                       da_description_sessions dese1,
                       da_described_objects deob2,
                       da_description_sessions dese2
                 where dese1.database_nm = vDatabase1
                   and dese1.schema_nm = vSchemaName
                   and dese1.object_type = vObjectType
                   and dese1.current_version_yn = 'Y'
                   and deob1.description_session_id = dese1.description_session_id
                   and dese2.database_nm = vDatabase2
                   and dese2.schema_nm = vSchemaName
                   and dese2.object_type = vObjectType
                   and dese2.current_version_yn = 'Y'
                   and deob2.description_session_id = dese2.description_session_id
                   and deob1.object_nm = deob2.object_nm
                   and deob1.child_item_count <> deob2.child_item_count
                 order by deob1.object_nm
               ) loop
         nDifference3 := nDifference3 + 1;

         vPadStr1 := fvGetStrPadChars(' ', 31 - length(x.object_nm));
         vPadStr2 := fvGetStrPadChars(' ', 4 - length(x.count1));
         vPadStr3 := fvGetStrPadChars(' ', 4 - length(x.count2));

         dbms_output.put_line('... ' || vObjectTypeMixedCase || ' ' || x.object_nm || vPadStr1 || 
                                 vPadStr2 || x.count1 || ' ' || tSubObjectName || 's in ' || 
                                 vDatabase1 || ', and ' || vPadStr3 || x.count2 || ' in ' || vDatabase2 || '.');

      end loop;

      if nDifference3 = 0 then
         dbms_output.put_line('... None found.');
      end if;

      dbms_output.put_line(vDivider2);

   end if;

   if vObjectType in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION', 'SEQUENCE') then

      -- Retrieve the grants associated with the object type in the penultimate database specified.

      nSectionNo := nSectionNo + 1;
      dbms_output.put_line(to_char(nSectionNo, 'RN') || ') ' || vObjectTypeMixedCase || 's in ' || vSchemaName || '@' || vDatabase1 || 
                              ' with differing grants than ' || vSchemaName || '@' || vDatabase2);

      for x in (select deob.object_nm, deog.grantor, deog.grantee, deog.privilege 
                  from da_described_object_grants deog,
                       da_described_objects deob,
                       da_description_sessions dese
                 where dese.database_nm = vDatabase1
                   and dese.schema_nm = vSchemaName
                   and dese.object_type = vObjectType
                   and dese.current_version_yn = 'Y'
                   and deob.description_session_id = dese.description_session_id
                   and deog.described_object_id = deob.described_object_id
                 order by object_nm, grantor
               ) loop

         begin

            select deob.object_nm
              into tObjectName
              from da_described_object_grants deog,
                   da_described_objects deob,
                   da_description_sessions dese
             where dese.database_nm = vDatabase2
               and dese.schema_nm = vSchemaName
               and dese.object_type = vObjectType
               and deob.object_nm = x.object_nm 
               and deog.grantor = x.grantor 
               and deog.grantee = x.grantee
               and deog.privilege = x.privilege
               and dese.current_version_yn = 'Y'
               and deob.description_session_id = dese.description_session_id
               and deog.described_object_id = deob.described_object_id
               and rownum = 1;

         exception

            when no_data_found then
               if vLastPrintedObject <> x.object_nm then
                  dbms_output.put_line('... FOUND: ' || vObjectTypeMixedCase || ' ' || x.object_nm);
                  nDifference4 := nDifference4 + 1;
                  vLastPrintedObject := x.object_nm;
               end if;

            when others then
               dbms_output.put_line('ERROR1: ' || x.object_nm || ' - ' || sqlerrm || '.');
               raise;
         end;

      end loop;

   end if;

   dbms_output.put_line(vDivider2);
   dbms_output.put_line('ANALYSIS COMPLETE'); 
   dbms_output.put_line(vDivider1);

   dbms_output.put_line('SUMMARY OF ' || vObjectType || 'S IN ' || vDatabase1 || ' AND ' || vDatabase2);

   if vObjectType in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION', 'SEQUENCE') then
      dbms_output.put_line('... ' || vObjectTypeMixedCase || 's found in ' || vDatabase1 || ' but not in ' || vDatabase2 || 
                              ': ' || tOutputSpacer || to_char(nDifference1, vCountNumFmt));
   end if;

   if vObjectType in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION', 'SEQUENCE') then
      dbms_output.put_line('... ' || vObjectTypeMixedCase || 's found in ' || vDatabase2 || ' but not in ' || vDatabase1 || 
                              ': ' || tOutputSpacer || to_char(nDifference2, vCountNumFmt));
   end if;

   if vObjectType in ('TABLE', 'VIEW', 'PACKAGE') then
      dbms_output.put_line('... Differing ' || tSubObjectName || ' counts in ' || vDatabase1 || ' vs ' || vDatabase2 || 
                              ':  ' || to_char(nDifference3, vCountNumFmt));
   end if;

   if vObjectType in ('TABLE', 'VIEW', 'PACKAGE', 'PROCEDURE', 'FUNCTION', 'SEQUENCE') then
      dbms_output.put_line('... Differing grants in ' || vDatabase1 || ' vs ' || vDatabase2 || 
                              ':         ' || to_char(nDifference4, vCountNumFmt));
   end if;

   dbms_output.put_line(vDivider1);

exception

   when excIncompleteParams then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' not specified.');

   when others then
      dbms_output.put_line('ERROR: ' || sqlerrm || '.');

end prCompareEnvironments;

procedure prAuditObjectNamingStandard   (pvDatabaseName                 IN     varchar2,
                                         pvSchemaName                   IN     varchar2,
                                         pvObjectType                   IN     varchar2)
is

   vDatabaseName                  varchar2(100) := upper(ltrim(rtrim(pvDatabaseName)));
   vSchemaName                    varchar2(100) := upper(ltrim(rtrim(pvSchemaName)));
   vObjectType                    varchar2(100) := upper(ltrim(rtrim(pvObjectType)));

   nCount                         number        := 0;
   nViolations                    number        := 0;
   vErrorStr                      varchar2(200) := '';
   vErrorSep                      varchar2(2)   := '';
   vObjectTypeMixedCase           varchar2(128);

   vLineTxt                       varchar2(100);
   
   excNoObjectTypeFound           exception;
   excIncompleteParams            exception;
   excBadParamValue               exception;

begin

   if (pvDatabaseName is null or pvSchemaName is null or pvObjectType is null) then

      if pvDatabaseName is null then
         vErrorStr := vErrorStr || vErrorSep || 'DATABASE_NAME';
         vErrorSep := ', ';
      end if;
      if pvSchemaName is null then
         vErrorStr := vErrorStr || vErrorSep || 'SCHEMA_NAME';
         vErrorSep := ', ';
      end if;
      if pvObjectType is null then
         vErrorStr := vErrorStr || vErrorSep || 'OBJECT_TYPE';
         vErrorSep := ', ';
      end if;

      raise excIncompleteParams;

   end if;

   if not (    fbValidDatabaseName(vDatabaseName)
           and fbValidSchemaName(vSchemaName)
           and fbValidObjectType(vObjectType)
          ) then

      if not fbValidDatabaseName(vDatabaseName) then
         vErrorStr := vErrorStr || vErrorSep || 'DATABASE_NAME: ' || vDatabaseName;
         vErrorSep := ', ';
      end if;
      if not fbValidSchemaName(vSchemaName) then
         vErrorStr := vErrorStr || vErrorSep || 'SCHEMA_NAME: ' || vSchemaName;
         vErrorSep := ', ';
      end if;
      if not fbValidObjectType(vObjectType) then
         vErrorStr := vErrorStr || vErrorSep || 'OBJECT_TYPE: ' || vObjectType;
         vErrorSep := ', ';
      end if;

      raise excBadParamValue;

   end if;

   begin
      select object_type_mixed_case_nm 
        into vObjectTypeMixedCase
        from da_object_types
       where object_type = vObjectType;
   exception
      when no_data_found then
         raise excNoObjectTypeFound;
   end;

   dbms_output.put_line(vDivider1);
   dbms_output.put_line('AUDITING ' || vObjectTypeMixedCase || ' Naming Standard in ' || vSchemaName || 
                           '@' || vDatabaseName);
   dbms_output.put_line(vDivider1);
   dbms_output.put_line(vDivider2);

   for x in (select do.object_nm
               from da_description_sessions ds,
                    da_described_objects do
              where ds.description_session_id = do.description_session_id
                and ds.current_version_yn = 'Y'
                and ds.database_nm = vDatabaseName
                and ds.schema_nm = vSchemaName
                and ds.object_type = vObjectType
              order by 1
            ) loop
      nCount := nCount + 1;

      if vObjectType in ('TABLE', 'SEQUENCE') then
         if (x.object_nm like vSchemaName || '\_%' escape '\') or
            REGEXP_LIKE(x.object_nm, '^[[:alnum:]]{2}\_') then
            null; -- Valid naming standard
         else
            vLineTxt := '... VIOLATION: ' || x.object_nm;
            dbms_output.put_line(vLineTxt);
            nViolations := nViolations + 1;
         end if;
      else
         null; -- Add rules for other object types here in the future
      end if;

   end loop;

   dbms_output.put_line(vDivider2);

   if nCount > 0 then
      vLineTxt := 'Audited ' || nCount || ' ' || lower(vObjectTypeMixedCase) || 's in ' || vSchemaName || '@' || vDatabaseName || '. Found ' || nViolations || ' violations.';
   else
      vLineTxt := 'Found zero ' || lower(vObjectTypeMixedCase) || 's to audit in ' || vSchemaName || '@' || vDatabaseName || '.';
   end if;

   dbms_output.put_line(vLineTxt);
   dbms_output.put_line(vDivider1);

exception

   when excIncompleteParams then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' was/were not specified.');

   when excBadParamValue then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' is/are not valid.');

   when excNoObjectTypeFound then
      dbms_output.put_line('ERROR: Invalid Object Type: ' || vObjectType);

   when others then
      dbms_output.put_line('ERROR: ' || sqlerrm || '.');

end prAuditObjectNamingStandard;

      end if;

      raise excBadParamValue;

   end if;

   for x in (select do.object_nm, do.object_created_dtm, do.object_last_ddl_dtm, dtc.*

                    da_described_objects do,
                    da_described_table_columns dtc

                and do.described_object_id = dtc.described_object_id
              order by dtc.column_id
            ) loop

         dbms_output.put_line(vDivider1);

         dbms_output.put_line('Name                           Null?    Type');
         dbms_output.put_line('------------------------------ -------- ---------------------');

      end if;

      vLineTxt := fvPaddedText(x.column_nm, 30);
      if x.nullable_yn = 'Y' then
         vLineTxt := vLineTxt || '          ';
      else
         vLineTxt := vLineTxt || ' NOT NULL ';
      end if;

         vLineTxt := vLineTxt || x.data_type;

         else
            vLineTxt := vLineTxt || x.data_precision || ',' || x.data_scale || ')';
         end if;
      else
         vLineTxt := vLineTxt || 'OTHER: ' || x.data_type;
      end if;

      dbms_output.put_line(vLineTxt);
      nCount := nCount + 1;

   end loop;

   dbms_output.put_line(vDivider2);
   dbms_output.put_line('...... Found ' || nCount || ' columns.');
   dbms_output.put_line(vDivider1);

exception

                and ds.database_nm = nvl(pvDatabaseName, ds.database_nm)
                and ds.schema_nm = nvl(pvSchemaName, ds.schema_nm)
                and ds.description_session_id = do.description_session_id
                and do.described_object_id = dtc.described_object_id
                and dtc.column_nm like '%' || vFieldName || '%'
              order by ds.database_nm, ds.schema_nm, do.object_nm, dtc.column_nm
            ) loop

      vLineTxt := '... ' || fvPaddedText(x.schema_nm || '.' || x.object_nm || '@' || x.database_nm, 45) 
                     || '   ' || fvPaddedText(x.column_nm, 30) || ' ';

      if x.data_type in ('BLOB', 'DATE', 'CLOB') then
         vLineTxt := vLineTxt || x.data_type;
      elsif x.data_type in ('VARCHAR2', 'CHAR') then
         vLineTxt := vLineTxt || x.data_type || '(' || x.data_length || ')';
      elsif x.data_type = 'NUMBER' then
         vLineTxt := vLineTxt || 'NUMBER(';
         if nvl(x.data_scale, 0) = 0 then
            vLineTxt := vLineTxt || x.data_precision || ')';
         else
            vLineTxt := vLineTxt || x.data_precision || ',' || x.data_scale || ')';
         end if;
      else
         vLineTxt := vLineTxt || 'OTHER: ' || x.data_type;
      end if;

      if x.nullable_yn = 'Y' then
         vLineTxt := vLineTxt || '          ';
      else
         vLineTxt := vLineTxt || ' NOT NULL ';
      end if;

      dbms_output.put_line(vLineTxt);

      nCount := nCount + 1;

   end loop;

   dbms_output.put_line('...... Found ' || nCount || ' tables.');

exception

   when excIncompleteParams then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' was/were not specified.');

   when excBadParamValue then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' is/are not valid.');

   when others then
      dbms_output.put_line('ERROR: ' || sqlerrm || '.');

end prWhereIsField;

procedure prWhereAreFields              (pvField1Name                   IN     varchar2,
                                         pvField2Name                   IN     varchar2,
                                         pvDatabaseName                 IN     varchar2 := null,
                                         pvSchemaName                   IN     varchar2 := null
                                         )
is

   vField1Name                    varchar2(100) := upper(ltrim(rtrim(pvField1Name)));
   vField2Name                    varchar2(100) := upper(ltrim(rtrim(pvField2Name)));
   vDatabaseName                  varchar2(100) := upper(ltrim(rtrim(pvDatabaseName)));
   vSchemaName                    varchar2(100) := upper(ltrim(rtrim(pvSchemaName)));

   nCount                         number        := 0;
   vErrorStr                      varchar2(200) := '';
   vErrorSep                      varchar2(2)   := '';
   
   vLineTxt                       varchar2(100);
   
   excIncompleteParams            exception;
   excBadParamValue               exception;

begin

   if (vField1Name is null or vField2Name is null) then

      if vField1Name is null then
         vErrorStr := vErrorStr || vErrorSep || 'FIELD1_NAME';
         vErrorSep := ', ';
      end if;

      if vField2Name is null then
         vErrorStr := vErrorStr || vErrorSep || 'FIELD2_NAME';
         vErrorSep := ', ';
      end if;

      raise excIncompleteParams;

   end if;

   if (   (    vDatabaseName is not null
           and not fbValidDatabaseName(vDatabaseName)
          )
       or (    vSchemaName is not null
           and not fbValidSchemaName(vSchemaName)
          )
      ) then

      if vDatabaseName is not null and not fbValidDatabaseName(vDatabaseName) then
         vErrorStr := vErrorStr || vErrorSep || 'DATABASE_NAME: ' || vDatabaseName;
         vErrorSep := ', ';
      end if;
      if vSchemaName is not null and not fbValidSchemaName(vSchemaName) then
         vErrorStr := vErrorStr || vErrorSep || 'SCHEMA_NAME: ' || vSchemaName;
         vErrorSep := ', ';
      end if;
      raise excBadParamValue;
   end if;


   for x in (select ds.database_nm, ds.schema_nm, 
                    do.object_nm, 
                    dtc1.column_nm "COL1_NM", dtc1.data_type "COL1_DATA_TYPE", dtc1.data_length "COL1_DATA_LENGTH",
                    dtc1.data_precision "COL1_DATA_PRECISION", dtc1.data_scale "COL1_DATA_SCALE", dtc1.nullable_yn "COL1_NULLABLE_YN",
                    dtc2.column_nm "COL2_NM", dtc2.data_type "COL2_DATA_TYPE", dtc2.data_length "COL2_DATA_LENGTH",
                    dtc2.data_precision "COL2_DATA_PRECISION", dtc2.data_scale "COL2_DATA_SCALE", dtc2.nullable_yn "COL2_NULLABLE_YN"
               from da_description_sessions ds,
                    da_described_objects do,
                    da_described_table_columns dtc1,
                    da_described_table_columns dtc2
              where ds.current_version_yn = 'Y'
                and ds.database_nm = nvl(pvDatabaseName, ds.database_nm)
                and ds.schema_nm = nvl(pvSchemaName, ds.schema_nm)
                and ds.description_session_id = do.description_session_id
                and do.described_object_id = dtc1.described_object_id
                and do.described_object_id = dtc2.described_object_id
                and dtc1.column_nm like '%' || vField1Name || '%'
                and dtc2.column_nm like '%' || vField2Name || '%'
                and dtc1.described_table_column_id < dtc2.described_table_column_id
              order by ds.database_nm, ds.schema_nm, do.object_nm, dtc1.column_nm
            ) loop

      vLineTxt := '... ' || fvPaddedText(x.schema_nm || '.' || x.object_nm || '@' || x.database_nm, 60);
      dbms_output.put_line(vLineTxt);

      vLineTxt := '... ... ' || fvPaddedText(x.col1_nm, 30) || '  ';
      if x.col1_data_type in ('BLOB', 'DATE', 'CLOB') then
         vLineTxt := vLineTxt || x.col1_data_type;
      elsif x.col1_data_type in ('VARCHAR2', 'CHAR') then
         vLineTxt := vLineTxt || x.col1_data_type || '(' || x.col1_data_length || ')';
      elsif x.col1_data_type = 'NUMBER' then
         vLineTxt := vLineTxt || 'NUMBER(';
         if nvl(x.col1_data_scale, 0) = 0 then
            vLineTxt := vLineTxt || x.col1_data_precision || ')';
         else
            vLineTxt := vLineTxt || x.col1_data_precision || ',' || x.col1_data_scale || ')';
         end if;
      else
         vLineTxt := vLineTxt || 'OTHER: ' || x.col1_data_type;
      end if;

      if x.col1_nullable_yn = 'Y' then
         vLineTxt := vLineTxt || '          ';
      else
         vLineTxt := vLineTxt || ' NOT NULL ';
      end if;

      dbms_output.put_line(vLineTxt);

      vLineTxt := '... ... ' || fvPaddedText(x.col2_nm, 30) || '  ';
      if x.col2_data_type in ('BLOB', 'DATE', 'CLOB') then
         vLineTxt := vLineTxt || x.col2_data_type;
      elsif x.col2_data_type in ('VARCHAR2', 'CHAR') then
         vLineTxt := vLineTxt || x.col2_data_type || '(' || x.col2_data_length || ')';
      elsif x.col2_data_type = 'NUMBER' then
         vLineTxt := vLineTxt || 'NUMBER(';
         if nvl(x.col2_data_scale, 0) = 0 then
            vLineTxt := vLineTxt || x.col2_data_precision || ')';
         else
            vLineTxt := vLineTxt || x.col2_data_precision || ',' || x.col2_data_scale || ')';
         end if;
      else
         vLineTxt := vLineTxt || 'OTHER: ' || x.col2_data_type;
      end if;

      if x.col2_nullable_yn = 'Y' then
         vLineTxt := vLineTxt || '          ';
      else
         vLineTxt := vLineTxt || ' NOT NULL ';
      end if;

      dbms_output.put_line(vLineTxt);

      nCount := nCount + 1;

   end loop;

   dbms_output.put_line('...... Found ' || nCount || ' tables.');

exception

   when excIncompleteParams then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' was/were not specified.');

   when excBadParamValue then
      dbms_output.put_line('ERROR: Parameter(s) ' || vErrorStr || ' is/are not valid.');

   when others then
      dbms_output.put_line('ERROR: ' || sqlerrm || '.');

end prWhereAreFields;

end PKG_DA_DESCRIBE_KW;
